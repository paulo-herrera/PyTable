  63    def add(self, name: str = None, data = None, allowRepetition = False):
  64        """ Adds a column to this table.
  65            
  66            Args:
  67                name: column id. If not passed, then column is assigned a generic id,
  68                      e.g col1
  69                data: a list of data or Column to be added as data to the new column. 
  70                allowRepetition: if True, columns with similar name can be stored in this table.
  71            
  72            Returns:
  73                This table.
  74        """
  75        name = name if name else "col%02d"%len(self.cols)
  76        
  77        if self.has(name) and not allowRepetition:
  78            assert False, "name is already in table: " + name
  79        
  80        if data and (isinstance(data, Column)): # and data == None:
  81            data.name = name
  82            self.cols.append(data)    
  83        elif data:
  84            c = Column(name)
  85            c.addData(data)
  86            self.cols.append(c)
  87        else:
  88            c = Column(name)
  89            self.cols.append(c)
  90        
  91        if data:
  92            self.max_rows = self.max_rows if (self.max_rows >= len(data)) else len(data)
  93        
  94        return self

 105    def append(self, other):
 106        """ Appends columns of other table to *this* table, 
 107            i.e. makes the columns of this table longer by appending data 
 108            from the other table.
 109            Both tables must have the same number of columns.
 110        
 111            Args:
 112                other: Table to append to this table.
 113                
 114            Returns:
 115                This table with appended values.
 116            
 117            **Note:** If you need to get a new table by merging this and other, 
 118                      clone this table first and then append other.
 119        """
 120        assert (isinstance(other,Table))
 121        assert len(self) == len(other)
 122        
 123        for i in range(len(self)):
 124            ic = self.cols[i]
 125            oc = other.cols[i]
 126            ic.addData(oc.data)   # type is checked internally
 127        
 128        self.__setMaxRows()
 129        return self

 153    def clone(self, shallow = True, newName = None):
 154        """ Creates a shallow or deep copy of this table.
 155            
 156            Args:
 157                shallow: if True, only pass references to columns in this table to new
 158                         table. If False, then creates a deep copy, so that this table
 159                         and new table do not share information.
 160                newName: if present, then use it as title of new table.
 161            
 162            Returns:
 163                New table.
 164        """
 165        newName = self.name + "(Copy)" if not newName else newName 
 166        t = Table(newName)
 167        for col in self.cols:
 168            ncol = col if shallow else col.clone() 
 169            t.add(ncol.name, ncol)
 170        return t

 175        """ Reads table from HDF5 file saved by calling toH5 or with a similar format.
 176            
 177            Args:
 178                root: 
 179            Returns:
 180                Read table and full path to file, (t, f).
 181        """

 185        import h5py
 186        
 187        if verbose: 
 188            print("Reading table from: " + src)
 189            
 190        src = os.path.abspath(src)
 191        h5 = h5py.File(src, "r")
 192        
 193        root = root if root else "/"
 194        g = h5[root]
 195        name = g.attrs["table"]
 196        date = g.attrs["date"]
 197        if verbose:
 198            print(" Table: " + name)
 199            print("   Saved on: " + date)
 200        
 201        t = Table(name)
 202        for name in g:
 203            #print(name)       # DEBUG
 204            ds = h5[root + "/" + name]
 205            id = name
 206            nvals = len(ds)
 207            dtype = str(ds.dtype)
 208            if verbose: print("   DATASET<%s>: %d values, type: %s"%(name, nvals, dtype))
 209            
 210            # Check if it creates too many temporaries
 211            a = ds[:]
 212            if "S" in dtype:            # strings are passed as raw binary, need to decode
 213                data = [e.decode("utf-8") for e in a]
 214            else:
 215                data = a.tolist()
 216            t.add(id, data)
 217            
 218        h5.close()
 219        if verbose: print("   Finished reading table")
 220        return t, src

 224    def has(self, key):
 225        """ Given a key returns True if it is in list of columns. 
 226            
 227            NOTE: keys are compared always as lower case. Thus, 
 228                 "Duck" == "duck" == "DUCK"
 229                 
 230            Args:
 231                key: an integer number or a name. 
 232                     It should be faster to call with key as an integer.
 233            
 234            Returns:
 235                True or false.
 236        """
 237        assert not is_iterable(key), key
 238        
 239        if (isinstance(key, int)):
 240            return (key >= 0 and key < len(self.cols))
 241            
 242        elif (isinstance(key, str)):
 243            ids = self.names(case = "L")
 244            return key.lower() in ids
 245            
 246        else:
 247            assert False, "Unknown type for key: " + str(key)

 319    def names(self, case = "M"):
 320        """ Returns a list with ids (names) of columns in this table. Depending on
 321            case, names are returned as lower, upper or a mix of lower and upper case.
 322            
 323            Args: 
 324                case: L (lower), U (upper) or M (mix lower and upper) [DEFAULT = M]
 325        """
 326        if case == "M":
 327            names = [c.name for c in self.cols]
 328        elif case == "L":
 329            names = [c.name.lower() for c in self.cols]
 330        elif case == "U":
 331            names = [c.name.upper() for c in self.cols]
 332        
 333        return names

 375    def convert(self, cols: List[int], types: List[str], fmt_date = None):
 376        """ Attempt to convert each column of this table to the specified type provided in the list fmt.
 377            
 378            Args:
 379                cols: list with columns indexes that should be converted. ç
 380                      if empty, assumes all columns in the table should be converted.
 381                types: a list with single characters that specify the new type of each column, 
 382                     i.e.["i", "f", "d", "s"].
 383                     If shorter than cols, then the last element is repeated.
 384                fmt_date: format used to convert strings to datetime objects.
 385                          Only needed if converting to dates.
 386            Returns:
 387                This Table.
 388                
 389        # """
 390        for t in types:
 391            assert t in ALLOWED_TYPES, t
 392        
 393        if len(cols) == 0: cols = [i for i in range(self.ncols())]
 394        
 395        if len(types) < len(cols):
 396            missing = len(cols) - len(types)
 397            append = missing * [types[-1]]
 398            types = types + append
 399        elif len(types) == len(cols):
 400            pass
 401        else:
 402            assert False
 403        
 404        for i in range(len(cols)):
 405            idx = cols[i]
 406            c = self.cols[idx]
 407            nt = types[i]
 408            if (c.type == "s") and (nt == "d"):
 409                assert fmt_date
 410                c1 = c.convert(nt , fmt_date)
 411            else:
 412                c1 = c.convert(nt) # using the default format
 413            self.cols[idx] = c1
 414        
 415        return self

 419    def print(self, maxRows: int = -1, writeTitle: bool=True, out = sys.stdout, \
 420              sep: str="\t", columnWidth: int=10, missing: str="-", verbose: bool=False, \
 421              start: int=0, lineBelow = True):
 422        """ Prints each elements in columns of this table in tabular format. 
 423            
 424            Args:
 425                maxRows: maximum number of rows to be printed. 
 426                         If <0, then all rows are printed.
 427                writeTitle: If true, prints name of table.
 428                out: stream where data should be printed.
 429                sep: string used as separator between columns.
 430                columnWidth: default width used to print columns. 
 431                             There is no guarantee that the formats used to print
 432                             elements of column fit withing this width.                              
 433                missing: string used to represent missing values in table. DEFAULT: "-"
 434                verbose: if True, then prints some additional information to sys.stdout.
 435                start: start printing at this row.
 436                lineBelow: if True, prints a line below column headers.
 437                
 438            Returns:
 439                Stream where data was printed. 
 440        """
 441        _fmt = "%" + str(columnWidth) + "s"
 442
 443        nchars = columnWidth * len(self.cols) 
 444        if writeTitle:
 445            out.write("=" * nchars + "\n")
 446            out.write("Table: %s\n" % (self.name))
 447            out.write("-" * nchars + "\n")
 448
 449        for c in self.cols:
 450            out.write( _fmt%c.name[:columnWidth] ) # TODO: Optimize
 451            out.write(sep)
 452        out.write("\n")
 453        if lineBelow: 
 454            out.write("-" * nchars + "\n")
 455    
 456        nrows = maxRows if (maxRows > 0 and self.max_rows > maxRows) else self.max_rows 
 457        for r in range(start, nrows):
 458            for c in self.cols:
 459                if len(c) >r:
 460                    s = c.format(r)
 461                    out.write(_fmt%s)
 462                else:
 463                    out.write(_fmt%(missing) )
 464                out.write(sep)
 465            out.write("\n")
 466        
 467        return out

 573    def read(src: str, sep: str=",", header=1, removeEmptyColumn=True, \
 574             verbose=True, encoding: str = "utf-8", allowRepetition = True, skip = 0):
 575        """ Reads table from file.
 576            
 577            Args:
 578                src: path to file.
 579                sep: string that separates columns as a regex, e.g. \w for white space. 
 580                header: line number that contains header (0 or 1, DEFAULT = 1). 
 581                removeEmptyColumn: if True, check and removed columns that only have empty strings.
 582                                   needed for ill-formed files. [DEFAULT=False]
 583                                   For large tables, it may be faster to import them, 
 584                                   and remove specific columns later.
 585                verbose: if True, prints some additional information.
 586                encoding: string that indicates file encoding.
 587                allowRepetition: if True, allows columns with same header id.
 588                skip: number of lines at beginning of file that should be skipped, 
 589                      e.g. comment lines [DEFAULT = 0]. 
 590                
 591            Returns:
 592                A new table with data read from file and a list with skipped lines. 
 593        """
 594        assert header <= 1, header
 595
 596        if verbose:
 597            print("Reading table from: ")
 598            print("   " + src)
 599        
 600        strs, skipped = read_tab_file(src, sep, strip=True, verbose=False, encoding = encoding, skip=skip)
 601        nlines= len(strs)
 602        ncols = len(strs[0])
 603        if verbose:
 604            print("   Read %d lines"%nlines)
 605            print("   %d columns"%ncols)
 606            if skipped and len(skipped) >0: print("   >>SKIPPED<<" + skipped[0])          # version
 607        
 608        h = strs[0] if header == 1 else ["col%04d"%c for c in range(ncols)]
 609        t = Table(name = src)
 610        for hh in h: t.add(name = hh, data=[], allowRepetition=allowRepetition)
 611        
 612        for c in range(ncols):
 613            cdata = []
 614            for r in range(header, nlines):
 615                v = strs[r][c]
 616                cdata.append(v)
 617            t[c].addData(cdata)
 618        
 619        if removeEmptyColumn:
 620            for c in range(ncols - 1, -1, -1):
 621                col = t.cols[c]
 622                if col.isBlank(): t.pop(c)
 623                
 624        t.__setMaxRows()
 625        
 626        return t, skipped

 376    def convert(self, cols: List[int], types: List[str], fmt_date = None):
 377        """ Attempt to convert each column of this table to the specified type provided in the list fmt.
 378            
 379            Args:
 380                cols: list with columns indexes that should be converted. ç
 381                      if empty, assumes all columns in the table should be converted.
 382                types: a list with single characters that specify the new type of each column, 
 383                     i.e.["i", "f", "d", "s"].
 384                     If shorter than cols, then the last element is repeated.
 385                fmt_date: format used to convert strings to datetime objects.
 386                          Only needed if converting to dates.
 387            Returns:
 388                This Table.
 389                
 390        # """
 391        for t in types:
 392            assert t in ALLOWED_TYPES, t
 393        
 394        if len(cols) == 0: cols = [i for i in range(self.ncols())]
 395        
 396        if len(types) < len(cols):
 397            missing = len(cols) - len(types)
 398            append = missing * [types[-1]]
 399            types = types + append
 400        elif len(types) == len(cols):
 401            pass
 402        else:
 403            assert False
 404        
 405        for i in range(len(cols)):
 406            idx = cols[i]
 407            c = self.cols[idx]
 408            nt = types[i]
 409            if (c.type == "s") and (nt == "d"):
 410                assert fmt_date
 411                c1 = c.convert(nt , fmt_date)
 412            else:
 413                c1 = c.convert(nt) # using the default format
 414            self.cols[idx] = c1
 415        
 416        return self

 574    def read(src: str, sep: str=",", header=1, removeEmptyColumn=True, \
 575             verbose=True, encoding: str = "utf-8", allowRepetition = True, skip = 0):
 576        """ Reads table from file.
 577            
 578            Args:
 579                src: path to file.
 580                sep: string that separates columns as a regex, e.g. \w for white space. 
 581                header: line number that contains header (0 or 1, DEFAULT = 1). 
 582                removeEmptyColumn: if True, check and removed columns that only have empty strings.
 583                                   needed for ill-formed files. [DEFAULT=False]
 584                                   For large tables, it may be faster to import them, 
 585                                   and remove specific columns later.
 586                verbose: if True, prints some additional information.
 587                encoding: string that indicates file encoding.
 588                allowRepetition: if True, allows columns with same header id.
 589                skip: number of lines at beginning of file that should be skipped, 
 590                      e.g. comment lines [DEFAULT = 0]. 
 591                
 592            Returns:
 593                A new table with data read from file and a list with skipped lines. 
 594        """
 595        assert header <= 1, header
 596
 597        if verbose:
 598            print("Reading table from: ")
 599            print("   " + src)
 600        
 601        strs, skipped = read_tab_file(src, sep, strip=True, verbose=False, encoding = encoding, skip=skip)
 602        nlines= len(strs)
 603        ncols = len(strs[0])
 604        if verbose:
 605            print("   Read %d lines"%nlines)
 606            print("   %d columns"%ncols)
 607            if skipped and len(skipped) >0: print("   >>SKIPPED<<" + skipped[0])          # version
 608        
 609        h = strs[0] if header == 1 else ["col%04d"%c for c in range(ncols)]
 610        t = Table(name = src)
 611        for hh in h: t.add(name = hh, data=[], allowRepetition=allowRepetition)
 612        
 613        for c in range(ncols):
 614            cdata = []
 615            for r in range(header, nlines):
 616                v = strs[r][c]
 617                cdata.append(v)
 618            t[c].addData(cdata)
 619        
 620        if removeEmptyColumn:
 621            for c in range(ncols - 1, -1, -1):
 622                col = t.cols[c]
 623                if col.isBlank(): t.pop(c)
 624                
 625        t.__setMaxRows()
 626        
 627        return t, skipped
